# SCDE

./build.sh

aufräumen:
./clean.sh

aufräumen+build:
./rebuild.sh


install preparations:
./prepare.sh





Bauen und starten:
./rebuild.sh
chmod a+x ~/LINUX/SCDE/dist/release/bin/SCDE
cd ..
cd bin
./SCDE
